<script type="module">
    // --- Game Data ---
    const foodList = [
        { name: "Apple", emoji: "🍎", good: true, info: "Apples are crunchy and clean your teeth!" },
        { name: "Candy", emoji: "🍭", good: false, info: "Sugar stays on teeth and causes cavities." },
        { name: "Milk", emoji: "🥛", good: true, info: "Milk has calcium to make teeth strong." },
        { name: "Soda", emoji: "🥤", good: false, info: "Soda has acid that hurts your tooth enamel." },
        { name: "Cheese", emoji: "🧀", good: true, info: "Cheese helps protect your teeth from acid." },
        { name: "Carrot", emoji: "🥕", good: true, info: "Carrots are full of fiber and good for gums." }
    ];

    let currentFoodIndex = 0;
    let canAnswer = false;

    // --- MediaPipe Setup ---
    import { FaceLandmarker, FilesetResolver } from "https://cdn.jsdelivr.net/npm/@mediapipe/tasks-vision/vision_bundle.js";

    let faceLandmarker;
    let video = document.getElementById("webcam");

    async function setupAI() {
        const filesetResolver = await FilesetResolver.forVisionTasks("https://cdn.jsdelivr.net/npm/@mediapipe/tasks-vision/wasm");
        faceLandmarker = await FaceLandmarker.createFromOptions(filesetResolver, {
            baseOptions: { modelAssetPath: `https://storage.googleapis.com/mediapipe-models/face_landmarker/face_landmarker/float16/1/face_landmarker.task` },
            runningMode: "VIDEO",
            numFaces: 1
        });
        startCamera();
    }

    function startCamera() {
        navigator.mediaDevices.getUserMedia({ video: true }).then((stream) => {
            video.srcObject = stream;
            video.addEventListener("loadeddata", predictWebcam);
            document.getElementById("explanation").innerText = "Ready! Tilt your head to start.";
            canAnswer = true;
        });
    }

    // --- Detection Logic ---
    async function predictWebcam() {
        if (faceLandmarker) {
            const startTimeMs = performance.now();
            const results = faceLandmarker.detectForVideo(video, startTimeMs);

            if (results.faceLandmarks && results.faceLandmarks.length > 0) {
                const landmarks = results.faceLandmarks[0];
                
                // Point 10 (Top of forehead) and Point 152 (Chin)
                const top = landmarks[10];
                const bottom = landmarks[152];

                // Calculate horizontal difference between top and bottom
                const tilt = top.x - bottom.x;

                // Thresholds: Because video is mirrored, top.x > bottom.x is a tilt to the player's LEFT
                if (canAnswer) {
                    if (tilt > 0.08) { // Tilt Left
                        checkAnswer(true);
                    } else if (tilt < -0.08) { // Tilt Right
                        checkAnswer(false);
                    }
                }
            }
        }
        window.requestAnimationFrame(predictWebcam);
    }

    // --- Game Logic ---
    function checkAnswer(isGood) {
        canAnswer = false; // Stop listening for tilts temporarily
        const food = foodList[currentFoodIndex];
        const feedbackDiv = document.getElementById("feedback");
        
        const isCorrect = (isGood === food.good);

        // Show Visual Feedback
        feedbackDiv.style.display = "block";
        feedbackDiv.innerText = isCorrect ? "✅ Correct!" : "❌ Try Again!";
        feedbackDiv.className = isCorrect ? "correct" : "wrong";

        document.getElementById("explanation").innerText = food.info;

        // Move to next food after 3 seconds
        setTimeout(() => {
            feedbackDiv.style.display = "none";
            currentFoodIndex = (currentFoodIndex + 1) % foodList.length;
            updateUI();
            canAnswer = true;
        }, 3000);
    }

    function updateUI() {
        const food = foodList[currentFoodIndex];
        document.getElementById("food-display").innerText = food.emoji;
        document.getElementById("food-name").innerText = food.name;
    }

    setupAI();
</script>
